Treatment 1: TWFE

#|include: FALSE
performance$treat1 = ifelse(is.na(performance$login), 0, 1)
performance$treat2 = ifelse(is.na(performance$qualified1_month), 0, 1)
month = as.data.frame(unique(performance$run_month))
colnames(month) = "run_month"
month_t = month %>% mutate(t = 1:nrow(month))
performance = performance %>% left_join(month_t, by = "run_month")
performance$E1 = 7
performance[performance$treat1 == 0, "E1"] =27
performance$qualified1_month = as.Date(performance$qualified1_month)
performance = performance %>% rename(time = t)
month_t$run_month = as.Date(month_t$run_month)
performance = performance %>%
  left_join(month_t, by = c("qualified1_month" = "run_month")) %>% rename(E2 = t)
performance[is.na(performance$E2), "E2"] = 27
performance$relative1 = performance$time - performance$E1
performance$relative2 = performance$time - performance$E2
performance = performance %>% group_by(player_id) %>% 
  mutate(best_time = best_time_imputed/max(best_time_imputed))
model1 = feols(best_time ~ D1|player_id + run_month, data = performance)

Conclusion

  • After the program took effect, players’ game performance improve 4% on average.

  • After players joined the incentive program there is an additional 4.3% increase in game performance
